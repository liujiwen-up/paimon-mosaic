Rust: Writing a File

Add mosaic-core as a dependency (path dependency within the workspace), then use MosaicWriter to create a file. Data is written as Arrow RecordBatch objects: 

use std::sync::Arc;
use arrow::datatypes::{DataType, Field, Schema};
use arrow::array::*;
use arrow::record_batch::RecordBatch;
use mosaic_core::writer::{MosaicWriter, WriterOptions};
use mosaic_core::spec::*;

// 1. Define an Arrow Schema
let arrow_schema = Schema::new(vec![
    Field::new("age", DataType::Int32, true),
    Field::new("name", DataType::Utf8, true),
    Field::new("score", DataType::Float64, true),
]);

// 2. Create writer (writes to any OutputFile implementation)
let mut writer = MosaicWriter::new(output, &arrow_schema, WriterOptions {
    num_buckets: 2,
    compression: COMPRESSION_ZSTD,
    ..Default::default()
})?;

// 3. Build an Arrow RecordBatch and write it
let ages = Int32Array::from((0..1000).map(|i| 20 + (i % 50)).collect::<Vec<_>>());
let names = StringArray::from((0..1000).map(|i| format!("user_{}", i)).collect::<Vec<_>>());
let scores = Float64Array::from((0..1000).map(|i| i as f64 * 1.5).collect::<Vec<_>>());
let batch = RecordBatch::try_new(
    Arc::new(Schema::new(vec![
        Field::new("age", DataType::Int32, true),
        Field::new("name", DataType::Utf8, true),
        Field::new("score", DataType::Float64, true),
    ])),
    vec![Arc::new(ages), Arc::new(names), Arc::new(scores)],
).unwrap();
writer.write_batch(&batch).unwrap();

// 4. Finalize
writer.close().unwrap();

// Estimated file size (for file rolling decisions)
let est = writer.estimated_file_size();

Writer Methods

MethodReturnDescription
write_batch(&RecordBatch)io::Result<()>Write an Arrow RecordBatch
estimated_file_size()u64Estimated output file size in bytes (for file rolling)
close()io::Result<()>Flush remaining data and write footer
num_row_groups()usizeNumber of row groups written (available after close)

Rust: Reading a File

MosaicReader is generic over any InputFile implementation. Implement the InputFile trait for your data source to read from local files, memory-mapped buffers, remote storage, etc.: 

use mosaic_core::reader::InputFile;

pub trait InputFile {
    fn read_at(&self, offset: u64, buf: &mut [u8]) -> io::Result<()>;
}

1. Open and Inspect the Schema

Pass your InputFile implementation and the file length directly to MosaicReader::new: 

use mosaic_core::reader::{MosaicReader, InputFile, ReaderAccess};

let reader = MosaicReader::new(my_input_file, file_len).unwrap();

// Iterate over all columns
for col in &reader.schema().columns {
    println!("name={} type={:?} nullable={}",
        col.name, col.data_type, col.nullable);
}

2. Read Row Groups as Arrow RecordBatch

Each row group is read as an Arrow RecordBatch via read_columns(). This returns fully-constructed Arrow arrays with proper null handling and type mapping. 

use arrow::array::*;

for rg_idx in 0..reader.num_row_groups() {
    let mut rg = reader.row_group_reader(rg_idx).unwrap();
    let batch = rg.read_columns().unwrap();

    println!("row group {} has {} rows, {} cols",
        rg_idx, batch.num_rows(), batch.num_columns());

    // Access columns by name
    let ages = batch.column_by_name("age").unwrap()
        .as_any().downcast_ref::<Int32Array>().unwrap();
    let names = batch.column_by_name("name").unwrap()
        .as_any().downcast_ref::<StringArray>().unwrap();

    for i in 0..batch.num_rows() {
        if !ages.is_null(i) {
            println!("age={} name={}", ages.value(i), names.value(i));
        }
    }
}

Projection Pushdown

Use row_group_reader_projected to read only specific columns. Only the buckets containing the projected columns are decompressed — reading 1 column out of 10,000 only touches 1 bucket instead of all 100. The returned batch contains only the projected columns. 

let name_col = reader.schema().columns.iter()
    .position(|c| c.name == "name").unwrap();
let score_col = reader.schema().columns.iter()
    .position(|c| c.name == "score").unwrap();

let mut rg = reader.row_group_reader_projected(rg_idx, &[name_col, score_col]).unwrap();
let batch = rg.read_columns().unwrap();
// batch contains only 2 columns: "name" and "score"
// Use batch.schema().field(i).name() to identify columns
Column ordering The reader preserves the original schema column order. Use reader.schema().columns[i].name or positional lookup to access columns.

Column Statistics (Filter Pushdown)

Enable per-column min/max statistics to allow query engines to skip entire row groups whose value range does not match a filter predicate.

Writing with Stats

let mut writer = MosaicWriter::new(output, &arrow_schema, WriterOptions {
    stats_columns: vec![0, 2],  // build stats for columns 0 and 2
    ..Default::default()
})?;

Getting Stats from Writer (after close)

writer.close()?;

for rg_idx in 0..writer.num_row_groups() {
    let stats = writer.row_group_stats(rg_idx);
    for stat in stats {
        println!("col={} nulls={} min={:?} max={:?}",
            stat.column_index, stat.null_count, stat.min, stat.max);
    }
}

Getting Row Group Row Count

for rg_idx in 0..reader.num_row_groups() {
    let num_rows = reader.row_group_num_rows(rg_idx)?;
    println!("row group {} has {} rows", rg_idx, num_rows);
}

Reading Stats from Reader

for rg_idx in 0..reader.num_row_groups() {
    let stats = reader.row_group_stats(rg_idx)?;
    for stat in stats {
        println!("col={} nulls={} min={:?} max={:?}",
            stat.column_index, stat.null_count, stat.min, stat.max);
    }
}

Each ColumnStats contains: column_index, null_count, min: Option<Value>, max: Option<Value>. When all values in the row group are null, min and max are None.
